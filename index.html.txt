function openMenu(name, emoji) {
  currentResto = name;

  document.getElementById('client-menu').style.display = 'block';

function backToHome() {
  document.getElementById('client-home').style.display = 'block';
  document.getElementById('client-menu').style.display = 'none';
  document.getElementById('client-tracking').style.display = 'none';
}
function renderMenu() {
  const el = document.getElementById('menu-items');
  el.innerHTML = menu.map(item => {
    const qty = cart[item.id] || 0;
    return `<div class="menu-item" style="display:flex;align-items:center;padding:12px 14px;background:var(--card2);border-radius:10px;border:1px solid var(--border);margin-bottom:8px">
      <span class="menu-emoji">${item.emoji}</span>
      <span class="menu-name">${item.name}</span>
      <span class="menu-price">${item.price.toLocaleString()} F</span>
      <button class="btn-add ${qty>0?'added':''}" onclick="addItem(${item.id})">${qty>0?'✓'+qty:'+'}</button>
    </div>`;
  }).join('');
  renderCart();
}
function addItem(id) {
  cart[id] = (cart[id] || 0) + 1;
  renderMenu();
}
function renderCart() {
  const items = menu.filter(m => cart[m.id]);
  const box = document.getElementById('cart-box');
  if (!items.length) { box.style.display='none'; return; }
  box.style.display = 'block';
  const rows = document.getElementById('cart-rows');
  let total = 0;
  rows.innerHTML = items.map(m => {
    const sub = m.price * cart[m.id];
    total += sub;
    return `<div class="cart-row"><span>${m.emoji} ${m.name} ×${cart[m.id]}</span><span style="color:var(--orange);font-weight:700">${sub.toLocaleString()} F</span></div>`;
  }).join('');
  document.getElementById('cart-total').innerHTML = `Total <span>${total.toLocaleString()} FCFA</span>`;
}
function placeOrder() {
  cart = {};
  document.getElementById('client-menu').style.display = 'none';
  document.getElementById('client-tracking').style.display = 'block';
  let prog = 0;
  renderTracking(prog);
  const iv = setInterval(() => {
    prog++;
    renderTracking(prog);
    if (prog === 2) document.getElementById('livreur-info').style.display = 'block';
    if (prog >= 3) { clearInterval(iv); showToast('🎉 Commande livrée ! Bon appétit !'); }
  }, 2000);
}
function renderTracking(prog) {
  const steps = [
    { icon:'✅', label:'Commande reçue' },
    { icon:'🍳', label:'En préparation' },
    { icon:'🏍️', label:'Livreur en route' },
    { icon:'🎉', label:'Livré !' },
  ];
  const el = document.getElementById('track-steps');
  el.innerHTML = steps.map((s, i) => {
    const done = i <= prog;
    const current = i === prog;
    return `<div style="display:flex;align-items:center;gap:14px">
      <div style="display:flex;flex-direction:column;align-items:center">
        <div style="width:36px;height:36px;border-radius:50%;background:${done?'var(--orange)':'var(--card2)'};border:2px solid ${done?'var(--orange)':'var(--border)'};display:flex;align-items:center;justify-content:center;font-size:16px;transition:all .5s">${done?s.icon:'○'}</div>
        ${i<3?`<div style="width:2px;height:22px;background:${i<prog?'var(--orange)':'var(--border)'};transition:background .5s"></div>`:''}
      </div>
      <div style="opacity:${done?1:.4};transition:opacity .5s">
        <div style="font-weight:${current?700:400}">${s.label}</div>
        ${current&&i<3?`<div style="font-size:11px;color:var(--orange)">En cours...</div>`:''}
      </div>
    </div>`;
  }).join('');
}
